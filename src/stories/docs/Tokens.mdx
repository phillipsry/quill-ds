# Tokens

Every Quill token, rendered live from the single source
(`src/tokens/quill.tokens.mjs`). Editing that file and running
`npm run build:tokens` regenerates the CSS, the shipped registry theme, the
Storybook manager theme, and the Figma (DTCG) export — this page updates
automatically because it imports the source directly.

export const Row = ({ name, value, figma, swatch }) => (
  <div style={{ display: 'flex', alignItems: 'center', gap: 12, padding: '8px 0', borderBottom: '1px solid rgba(42,38,34,0.08)', fontSize: 12 }}>
    {swatch}
    <code style={{ width: 200 }}>{name}</code>
    <span style={{ color: '#675F58', flex: 1 }}>{value}</span>
    {figma && <code style={{ color: '#944A33' }}>{figma}</code>}
  </div>
)

export const ColorSwatch = ({ v }) => (
  <span style={{ display: 'inline-flex', gap: 4 }}>
    <span style={{ width: 22, height: 22, borderRadius: 5, background: v.light, border: '1px solid rgba(42,38,34,0.15)' }} title={`light ${v.light}`} />
    <span style={{ width: 22, height: 22, borderRadius: 5, background: v.dark, border: '1px solid rgba(42,38,34,0.15)' }} title={`dark ${v.dark}`} />
  </span>
)

## Colors

export const colorRows = (obj, prefix = []) => Object.entries(obj).flatMap(([k, v]) =>
  ('light' in v && 'dark' in v)
    ? [<Row key={[...prefix, k].join('-')} swatch={<ColorSwatch v={v} />}
        name={'--' + [...prefix, k].filter(p => p !== 'base').join('-')}
        value={`${v.light} / ${v.dark}`} figma={`color/${[...prefix, k].join('/')}`} />]
    : colorRows(v, [...prefix, k])
)

<div>{colorRows(tokens.color)}</div>
